JavaScript Mastery Guide: From Beginner to Pro

This guide covers JavaScript from basics to advanced concepts with examples and projects. Open this file in a browser to interact with the code.

1. Beginner Level: Basics

Variables and Data Types

JavaScript variables store data. Use let, const, or var.

// Variables
let name = "John"; // String
const age = 25; // Number
var isStudent = true; // Boolean

// Data types: string, number, boolean, undefined, null, object, symbol
console.log(typeof name); // "string"

Operators

// Arithmetic
let sum = 5 + 3; // 8
let diff = 5 - 3; // 2

// Comparison
console.log(5 > 3); // true
console.log(5 === "5"); // false (strict equality)

// Logical
console.log(true && false); // false
console.log(true || false); // true

Control Structures

// If-else
if (age >= 18) {
    console.log("Adult");
} else {
    console.log("Minor");
}

// Loops
for (let i = 0; i < 5; i++) {
    console.log(i);
}

let count = 0;
while (count < 3) {
    console.log(count);
    count++;
}

2. Intermediate Level: Functions and Objects

Functions

// Function declaration
function greet(name) {
    return "Hello, " + name;
}

// Arrow function (ES6)
const greetArrow = (name) => "Hello, " + name;

// Function call
console.log(greet("Alice")); // "Hello, Alice"

Objects and Arrays

// Object
let person = {
    name: "Bob",
    age: 30,
    greet: function() {
        return "Hi, I'm " + this.name;
    }
};
console.log(person.greet()); // "Hi, I'm Bob"

// Array
let fruits = ["apple", "banana", "orange"];
fruits.push("grape");
console.log(fruits[1]); // "banana"

3. DOM Manipulation

The DOM is the structure of HTML elements. JavaScript can change it.

// Select elements
let button = document.getElementById("myButton");
let paragraphs = document.querySelectorAll("p");

// Change content
button.textContent = "Clicked!";

// Add event listener
button.addEventListener("click", function() {
    alert("Button clicked!");
});

4. Advanced Level: Closures, Prototypes, Async

Closures

Functions remember their outer scope.

function outer() {
    let count = 0;
    return function inner() {
        count++;
        return count;
    };
}
let counter = outer();
console.log(counter()); // 1
console.log(counter()); // 2

Prototypes and Classes

// Constructor function
function Person(name, age) {
    this.name = name;
    this.age = age;
}
Person.prototype.greet = function() {
    return "Hello, " + this.name;
};

// ES6 Class
class Animal {
    constructor(name) {
        this.name = name;
    }
    speak() {
        return this.name + " makes a sound";
    }
}

Asynchronous JavaScript

// Promises
let promise = new Promise((resolve, reject) => {
    setTimeout(() => resolve("Done!"), 1000);
});
promise.then(result => console.log(result));

// Async/Await
async function fetchData() {
    try {
        let response = await fetch("https://api.example.com/data");
        let data = await response.json();
        return data;
    } catch (error) {
        console.error(error);
    }
}

ES6+ Features

// Template literals
let name = "Alice";
console.log(`Hello, ${name}!`); // "Hello, Alice!"

// Destructuring
let [a, b] = [1, 2];
let {x, y} = {x: 10, y: 20};

// Spread operator
let arr1 = [1, 2, 3];
let arr2 = [...arr1, 4, 5]; // [1, 2, 3, 4, 5]

// Arrow functions (already shown)
const add = (a, b) => a + b;

Error Handling

try {
    let result = riskyOperation();
    console.log(result);
} catch (error) {
    console.error("An error occurred:", error.message);
} finally {
    console.log("This always runs");
}

Regular Expressions

// Create regex
let regex = /hello/i; // Case-insensitive
let regex2 = new RegExp("world", "g"); // Global

// Test and match
console.log(regex.test("Hello World")); // true
console.log("Hello World".match(regex)); // ["Hello"]

// Replace
let str = "Hello World";
console.log(str.replace(/world/i, "Universe")); // "Hello Universe"

Browser APIs

// Local Storage
localStorage.setItem("name", "John");
console.log(localStorage.getItem("name")); // "John"

// Geolocation
if (navigator.geolocation) {
    navigator.geolocation.getCurrentPosition(position => {
        console.log(position.coords.latitude, position.coords.longitude);
    });
}

// Timers
setTimeout(() => console.log("Delayed"), 1000);
let interval = setInterval(() => console.log("Repeated"), 500);
clearInterval(interval);

Modules (ES6)

// In module.js
export const PI = 3.14159;
export function circleArea(radius) {
    return PI * radius * radius;
}

// In main.js
import { PI, circleArea } from './module.js';
console.log(circleArea(5));
